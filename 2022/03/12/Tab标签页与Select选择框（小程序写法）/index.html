state = {
currentBoxId: ‘stepone’ //当前显示的View的id

switch (currentFlag) {
case ‘steponeNext’:
this.setState({
currentBoxId: ‘steptwo’

break;
case ‘steptwoPrev’:
this.setState({
currentBoxId: ‘stepone’
})
break;
case ‘steptwoNext’:
this.setState({
currentBoxId: ‘stepthree’
})
break;
case ‘stepthreePrev’:
this.setState({
currentBoxId: ‘steptwo’
})
break;
case ‘stepthreeNext’:
this.setState({
currentBoxId: ‘stepfour’
})
break;
case ‘stepfourPrev’:
this.setState({
currentBoxId: ‘stepthree’
})
break;
case ‘stepfourNext’:
this.setState({
currentBoxId: ‘stepfive’
})
break;
